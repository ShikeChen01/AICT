\section{Scope}
This document provides detailed implementation specifications for the minimal folder layout
listed in \texttt{docs/general\_tech\_spec.md}. It defines required imports, module exports, and
module-level responsibilities for each folder in the minimal layout.

\section{Conventions}
\begin{itemize}[noitemsep]
  \item All source files are TypeScript unless otherwise noted.
  \item Files under \texttt{src/shared/} must avoid runtime dependencies on VS Code APIs or webview globals.
  \item All exported symbols must be documented in the module spec sections below.
  \item Internal imports use workspace-relative paths (e.g., \texttt{src/shared/types/rpc}).
\end{itemize}

\section{Extension Host: \texttt{src/extension}}

\subsection{\texttt{src/extension/rpc}}
\paragraph{Purpose} Typed RPC server bridging the webview UI and extension host.

\paragraph{Required imports}
\begin{itemize}[noitemsep]
  \item External: \texttt{vscode}, \texttt{zod}
  \item Node: \texttt{node:events}, \texttt{node:crypto}, \texttt{node:timers}
  \item Internal: \texttt{src/shared/types/rpc}, \texttt{src/shared/schemas/rpcSchema},
        \texttt{src/extension/policy}, \texttt{src/extension/runner}, \texttt{src/extension/patch},
        \texttt{src/extension/repoIndex}, \texttt{src/extension/storage}, \texttt{src/extension/cloud}
\end{itemize}

\paragraph{Exported modules}
\begin{itemize}[noitemsep]
  \item \texttt{rpcServer.ts} -- Create and manage the RPC server lifecycle.
  \item \texttt{rpcHandlers.ts} -- Register RPC method handlers and dependency wiring.
  \item \texttt{rpcValidation.ts} -- Validate and coerce request/response payloads.
  \item \texttt{rpcTransport.ts} -- Webview transport adapter and message bridge.
\end{itemize}

\paragraph{Module specifications}
\subsubsection*{Module: \texttt{rpcServer.ts}}
\begin{itemize}[noitemsep]
  \item Exports: \texttt{createRpcServer(options)}, \texttt{RpcServer}, \texttt{RpcServerOptions}
  \item Responsibilities:
    \begin{itemize}[noitemsep]
      \item Listen on the transport and route requests to handlers.
      \item Track active requests and enforce timeouts.
      \item Emit server events for job status, logs, and repo index updates.
    \end{itemize}
  \item Inputs/Outputs: accepts \texttt{RpcTransport} and handler registry; emits \texttt{RpcResponse} and events.
  \item Failure modes: malformed payloads, unknown method, handler errors -> structured error response.
\end{itemize}

\subsubsection*{Module: \texttt{rpcHandlers.ts}}
\begin{itemize}[noitemsep]
  \item Exports: \texttt{RpcHandlerRegistry}, \texttt{registerCoreHandlers(deps)}
  \item Responsibilities: provide handlers for \texttt{startWork}, \texttt{runTests}, \texttt{applyPatch},
        \texttt{repoIndex}, \texttt{exportBundle}, and \texttt{cancelJob}.
  \item Dependencies: policy engine, runner, patch engine, repo index, storage, cloud gateway.
\end{itemize}

\subsubsection*{Module: \texttt{rpcValidation.ts}}
\begin{itemize}[noitemsep]
  \item Exports: \texttt{validateRpcRequest}, \texttt{validateRpcResponse}
  \item Responsibilities: validate payloads with Zod and return typed data or structured errors.
\end{itemize}

\subsubsection*{Module: \texttt{rpcTransport.ts}}
\begin{itemize}[noitemsep]
  \item Exports: \texttt{createWebviewTransport(webview)}, \texttt{RpcTransport}
  \item Responsibilities: wrap \texttt{webview.postMessage} and \texttt{onDidReceiveMessage} into
        a typed transport that assigns request ids.
\end{itemize}
